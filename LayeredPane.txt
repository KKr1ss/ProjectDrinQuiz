import com.k33ptoo.components.KGradientPanel;
import javax.swing.*;
import java.awt.*;

public class classHostFrame {

    public classHostFrame() {

        KGradientPanel panel1 = new KGradientPanel();
        panel1.setOpaque(true);
        panel1.setkStartColor(new java.awt.Color(87, 24, 69));
        panel1.setkEndColor(new java.awt.Color(199, 0, 57));
        panel1.setBounds(50, 50, 400, 200);

        KGradientPanel panel2 = new KGradientPanel();
        panel2.setOpaque(true);
        panel2.setkStartColor(new java.awt.Color(87, 24, 69));
        panel2.setkEndColor(new java.awt.Color(199, 0, 57));
        panel2.setBounds(100, 100, 400, 200);

        KGradientPanel panel3 = new KGradientPanel();
        panel3.setOpaque(true);
        panel3.setkStartColor(new java.awt.Color(87, 24, 69));
        panel3.setkEndColor(new java.awt.Color(199, 0, 57));
        panel3.setBounds(150, 150, 400, 200);

        JLayeredPane layeredPane = new JLayeredPane();
        layeredPane.setBounds(0, 0, 800, 600);

        //layeredPane.add(panel1, JLayeredPane.DEFAULT_LAYER);
        layeredPane.add(panel1, Integer.valueOf(0));
        layeredPane.add(panel2, Integer.valueOf(1));
        layeredPane.add(panel3, Integer.valueOf(2));

        JFrame frame = new JFrame("JLayeredPane");
        frame.add(layeredPane);
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(new Dimension(800, 600));
        frame.setLayout(null);
        frame.setVisible(true);
    }
}
